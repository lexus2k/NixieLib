57  Wire.write(0x00);
58  Wire.endTransmission();
59  Wire.requestFrom(I2C_ADDR_DS3231, 7);
60 
61  if (Wire.available())

66  m_day_of_week = Wire.read();

70  }

108 int16_t Ds3231::getTemp()
109 {
110  //temp registers (11h-12h) get updated automatically every 64s
111  Wire.beginTransmission(I2C_ADDR_DS3231);
112  Wire.write(0x11);
113  Wire.endTransmission();
114  Wire.requestFrom(I2C_ADDR_DS3231, 2);
115  if(Wire.available())
116  {
117  uint8_t temp = Wire.read();
118  return (((temp & 0x7F) << 2) | (Wire.read() >> 6)) * ((temp & 0x80) ? -1: 1);
119  }
120  else
121  return 0;
122 }
