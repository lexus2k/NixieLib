38 bool NixieTube::update()
39 {
40  /* To reduce load of uC, during active period perform calculations only during
41  updating pwm state
42  */
43  if (g_nixieUs - m_uts < CHANGE_INTERVAL)
44  {
45  if (g_nixieUs - m_uts < m_impulseWidth )
46  {
47  if ( (m_pinState & PIN_STATE_POWERED) && !(m_pinState & PIN_STATE_PWM_HIGH ) )
48  {
49  m_pinState |= PIN_STATE_PWM_HIGH;
50  updatePinState();
51  }
52  }
53  else if (m_pinState & PIN_STATE_PWM_HIGH)
54  {
55  m_pinState &= ~(PIN_STATE_PWM_HIGH);
56  updatePinState();
57  }
58  return false;
59  }
60  m_uts = g_nixieUs;
61 
62  switch (m_state)
63  {
64  case TUBE_OFF:
65  m_mts = g_nixieMs;
66  break;
67  case TUBE_SMOOTH_OFF:
68  if (m_tempBrightness == 0)
69  {
70  m_state = TUBE_OFF;
71  }
72  else
73  {
74  updateDimming();
75  }
76  break;
77  case TUBE_NORMAL:
78  m_mts = g_nixieMs;
79  m_digit = m_value;
80  updateDimming();
81  break;
82  case TUBE_BLINK_ON:
83  if ((g_nixieMs & (BLINK_INTERVAL - 1)) <= (BLINK_INTERVAL >> 1))
84  {
85  m_state = TUBE_BLINK_OFF;
86  }
87  m_digit = m_value;
88  m_mts = g_nixieMs;
89  updateDimming();
90  break;
91  case TUBE_BLINK_OFF:
92  if ((g_nixieMs & (BLINK_INTERVAL - 1)) > (BLINK_INTERVAL >> 1))
93  {
94  m_state = TUBE_BLINK_ON;
95  }
96  m_mts = g_nixieMs;
97  updateDimming();
98  break;
99  case TUBE_SCROLL_OFF:
100  if (g_nixieMs - m_mts >= SCROLL_INTERVAL)
101  {
102  if (m_delayedScroll)
103  {
104  if (g_nixieMs - m_mts < m_dimmingTs )
105  {
106  m_dimmingTs = m_dimmingTs - (g_nixieMs - m_mts);
107  }
108  else
109  {
110  m_dimmingTs = g_nixieMs;
111  m_delayedScroll = 0;
112  }
113  }
114  m_mts += SCROLL_INTERVAL;
115  if (++m_digit == 10)
116  {
117  m_digit = 0;
118  }
119  if (m_tempBrightness == 0)
120  {
121  m_state = m_next;
122  }
123  }
124  if (!m_delayedScroll) updateDimming();
125  break;
126  case TUBE_SCROLL_WRAP:
127  case TUBE_SCROLL_FORWARD:
128  if (g_nixieMs - m_mts >= SCROLL_INTERVAL)
129  {
130  m_mts += SCROLL_INTERVAL;
131  if (m_delayedScroll > 0) m_delayedScroll--;
132  if (++m_digit == 10)
133  {
134  m_digit = 0;
135  }
136  if ((m_digit == m_value) && (m_delayedScroll == 0))
137  {
138  m_state = m_next;
139  }
140  }
141  updateDimming();
142  break;
143  case TUBE_SCROLL_BACK:
144  if (g_nixieMs - m_mts >= SCROLL_INTERVAL)
145  {
146  m_mts += SCROLL_INTERVAL;
147  if (m_delayedScroll > 0) m_delayedScroll--;
148  if (m_digit-- == 0)
149  {
150  m_digit = 9;
151  }
152  if ((m_digit == m_value) && (m_delayedScroll == 0))
153  {
154  m_state = m_next;
155  }
156  }
157  updateDimming();
158  break;
159  case TUBE_SCROLL_ON:
160  if (g_nixieMs - m_mts < m_dimmingTs )
161  {
162  m_dimmingTs = m_dimmingTs - (g_nixieMs - m_mts);
163  }
164  else
165  {
166  m_dimmingTs = g_nixieMs;
167  scrollForward();
168  }
169  m_mts = g_nixieMs;
170  break;
171  default:
172  /* for unknown state return OFF */
173  m_digit = 0xFF;
174  break;
175  }
176  return true;
177 }

195 void NixieTube::updatePinState()
196 {
197  // Set digit code for new digit to display.
198  if ( (m_pinState & PIN_STATE_PWM_HIGH) && isBurning() )
199  {
200  m_driver->switchPin( m_map[m_digit] );
201  m_driver->setExtendedPins( m_flags );
202  // Turn ON next bulb
203  if ( m_pin != 0xFF )
204  {
205  nixiePinHigh( m_pin );
206  }
207  }
208  else
209  {
210  if ( m_pin != 0xFF )
211  {
212  nixiePinLow( m_pin );
213  }
214  }
215 }

218 void NixieTube::updateDimming ()
219 {
220  if (m_mts - m_dimmingTs >= DIMMING_INTERVAL)
221  {
222  m_dimmingTs += DIMMING_INTERVAL;
223  if ((m_state == TUBE_SMOOTH_OFF) || (m_state == TUBE_SCROLL_OFF))
224  {
225  if (m_tempBrightness > 0) setActiveBrightness( m_tempBrightness - 1 );
226  }
227  else if (m_tempBrightness != m_brightness)
228  {
229  setActiveBrightness( m_tempBrightness + (m_tempBrightness < m_brightness ? +1 : -1) );
230  }
231  }
232 }

274 void NixieTube::operator =(const NixieTube &tube)
275 {
276  /* Copy displayed value and current state of the tube digit */
277  m_value = tube.m_value;
278  m_digit = tube.m_digit;
279  /* Copy brightness */
280  m_brightness = tube.m_brightness;
281  m_impulseWidth = tube.m_impulseWidth;
282  m_tempBrightness = tube.m_tempBrightness;
283  /* Copy state variables */
284  m_state = tube.m_state;
285  m_next = tube.m_next;
286  /* Copy flags from the source tube */
287  m_flags = tube.m_flags;
288  m_delayedScroll = tube.m_delayedScroll;
289 
290  /* We do not copy timer related variables */
291  /* Each tube object must calculate them itself.
292  * Otherwise, it breaks tube effects. */
