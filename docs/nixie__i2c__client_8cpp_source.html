49 void SimpleClientI2c::requestEvent()
50 {
51  if ( (m_regPtr != nullptr) && (m_regIndex < m_regCount) )
52  {
53  I2CBUS_Send(m_regPtr[m_regIndex]);
54  m_regIndex++;
55  if (m_regIndex >= m_regCount)
56  {
57  m_regIndex = m_regCount - 1;
58  }
59  }
60  else
61  {
62  }
63 }

66 void SimpleClientI2c::receiveEvent(uint8_t howMany)
67 {
68  if (howMany < 1)
69  {
70  return;
71  }
72  m_regIndex = I2CBUS_Receive();
73  if ((m_regCount != 0) && (m_regIndex<m_regCount))
74  {
75  while (--howMany)
76  {
77  uint8_t data = I2CBUS_Receive();
78  if ( (m_regAccessPtr != nullptr) && (m_regAccessPtr[m_regIndex] == I2C_READ_WRITE) )
79  {
80  m_regPtr[m_regIndex] = data;
81  }
82  if (++m_regIndex >= m_regCount)
83  {
84  m_regIndex = m_regCount - 1;
85  }
86  }
87  }
88  else
89  {
90  while (--howMany) I2CBUS_Receive();
91  }
92 }
