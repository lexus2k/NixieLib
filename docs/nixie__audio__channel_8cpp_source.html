25 static inline long getNoteDuration(const uint8_t *ptr, EMelodyType type)
26 {
27  switch (type)
28  {
29  case MELODY_TYPE_PROGMEM_SAMPLING:
30  return 1000 * (long)pgm_read_word_near(ptr+2);
31  case MELODY_TYPE_PROGMEM_TEMPO:
32  return 1000000 / pgm_read_byte_near(ptr+2);
33  case MELODY_TYPE_SAMPLING_NOTE:
34  return 1000 * (long)( *((uint16_t *)(ptr+2)) );
35  case MELODY_TYPE_TEMPO_NOTE:
36  return 1000000 / (*(ptr+2));
37  default:
38  break;
39  }
40  return 0;
41 }

63 uint8_t NixieAudioChannel::update(unsigned long ts)
64 {
65  if (m_notes == nullptr)
66  {
67  m_ts = ts;
68  return 0xFF;
69  }
70  bool updated = false;
71  while (ts - m_ts >= m_halfPeriod)
72  {
73  m_ts += m_halfPeriod;
74  m_duration -= (long)m_halfPeriod;
75  if (m_totalDuration != 0)
76  {
77  m_totalDuration -= m_halfPeriod;
78  }
79  updated = true;
80  }
81  if (updated)
82  {
83  if (m_totalDuration < 0)
84  {
85  m_notes = nullptr;
86  if (m_pin != 0xFF) nixieDigitalWrite(m_pin, LOW);
87  return 0xFF;
88  }
89  switch (m_freq)
90  {
91  case NOTE_SILENT:
92  case NOTE_PAUSE:
93  break;
94  default:
95  m_pinState = !m_pinState;
96  break;
97  }
98  if (m_pin != 0xFF) nixieDigitalWrite(m_pin, m_pinState);
99  if (m_duration <= 0)
100  {
101  playNext();
102  }
103  }
104  return m_pinState;
105 }

109 void NixieAudioChannel::playNote()
110 {
111  m_freq = getNoteFreq(m_notes, m_pmode);
112  switch (m_freq)
113  {
114  case NOTE_STOP:
115  m_duration += 1000000;
116  m_halfPeriod = 10000;
117  m_notes = nullptr;
118  digitalWrite(m_pin, LOW);
119  break;
120  case NOTE_SILENT:
121  m_halfPeriod = 10000;
122  m_duration += getNoteDuration(m_notes, m_pmode);
123  break;
124  default:
125  m_halfPeriod = (1000000 >> 1) / m_freq;
126  m_duration += getNoteDuration(m_notes, m_pmode);
127  break;
128  }
129 }

133 void NixieAudioChannel::playNext()
134 {
135  if (m_freq == NOTE_PAUSE)
136  {
137  switch (m_pmode)
138  {
139  case MELODY_TYPE_PROGMEM_SAMPLING:
140  m_notes+=4;
141  break;
142  case MELODY_TYPE_PROGMEM_TEMPO:
143  m_notes+=3;
144  break;
145  case MELODY_TYPE_SAMPLING_NOTE:
146  case MELODY_TYPE_TEMPO_NOTE:
147  m_notes = nullptr;
148  digitalWrite(m_pin, LOW);
149  return;
150  default:
151  break;
152  }
153  }
154  else
155  {
156  m_freq = NOTE_PAUSE;
157  if (m_pause < 0)
158  m_halfPeriod = getNoteDuration(m_notes, m_pmode)*(-m_pause) >> 5;
159  else if (m_pause > 0)
160  m_halfPeriod = (long)m_pause * 1000;
161  m_duration += m_halfPeriod;
162  return;
163  }
164  playNote();
165 }
