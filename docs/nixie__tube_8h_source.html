46 // Do not use lower bit for invisible states
47 // Lower bit=1 means that tube is burning
48 typedef enum
49 {
50  TUBE_OFF = 0x00, // invisible
51  TUBE_NORMAL = 0x01, // visible
52  TUBE_SCROLL_WRAP = 0x03, // visible
53  TUBE_SCROLL_FORWARD = 0x05, // visible
54  TUBE_SCROLL_BACK = 0x07, // visible
55  TUBE_SMOOTH_OFF = 0x09, // visible
56  TUBE_BLINK = 0x0B, // visible
57  TUBE_BLINK_ON = TUBE_BLINK, // 0x0B visible
58  TUBE_BLINK_OFF = 0x0C, // invisible
59  TUBE_SCROLL_OFF = 0x0D, // visible
60  TUBE_SCROLL_ON = 0x0E, // invisible
61  TUBE_SMOOTH_ON = 0x0F, // visible
62 } ENixieTubeState;
