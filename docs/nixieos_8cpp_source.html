112 void switchTask(uint8_t id)
113 {
114  uint8_t i=0;
115  for(;;)
116  {
117  uint8_t nextId = pgm_read_byte(&s_tasks[i].id);
118  if (nextId == TASK_ID_INVALID)
119  {
120  break;
121  }
122  if (nextId == id)
123  {
124  s_activeTask.id = nextId;
125  s_activeTask.enterCb = pgm_read_ptr(&s_tasks[i].enterCb);
126  s_activeTask.stateCb = pgm_read_ptr(&s_tasks[i].stateCb);
127  s_activeTask.eventCb = pgm_read_ptr(&s_tasks[i].eventCb);
128  if (s_activeTask.enterCb)
129  {
130  s_activeTask.enterCb();
131  }
132  break;
133  }
134  i++;
135  };
136 }

223 void run()
224 {
225  uint8_t lms = (uint8_t)g_nixieMsEx;
226  g_nixieMsEx = millis();
227  g_nixieUsEx = micros();
228  /* Recalculate timers only when milliseconds counter changes */
229  if (lms != (uint8_t)g_nixieMsEx)
230  {
231  for(uint8_t id=0; id<NIXIEOS_MAX_TIMERS; id++)
232  {
233  if (s_timers[id].aimed)
234  {
235  if (g_nixieMsEx - s_timers[id].startMs > s_timers[id].timeoutMs)
236  {
237  s_timers[id].aimed = false;
238  NixieOs::sendEvent(NIXIEOS_EVENT_TIMEOUT, id);
239  }
240  }
241  }
242  }
243  if (s_events[0].event != 0)
244  {
245  SNixieEvent event = s_events[0];
246  for (uint8_t i=0; i<MAX_NIXIE_EVENTS - 1; i++)
247  {
248  s_events[i] = s_events[i+1];
249  }
250  s_events[MAX_NIXIE_EVENTS - 1].event = 0;
251  s_activeTask.eventCb(event);
252  }
253  s_activeTask.stateCb();
254 }
