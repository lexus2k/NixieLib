60 /*
61  * IN-12/IN-14 Datasheet.
62  * Impulse mode: 70us / refresh frequency 1-1.8kHz
63  */
64 
65 /* Burning time for each digit.
66  * When 1 digit is ON, other are OFF. This means that during full cycle
67  * bulb is OFF for CHANGE_INTERVAL*5.
68  * IMPULSE_WIDTH should be greater than 70us (according to IN-14 spec) and working
69  * frequency should be below 1-1.8kHz, otherwise the gas in the the tube will be visible.
70  * On other way, Atmega328 supports only 4us accuracy.
71  * Nixie Display module uses 32 levels of brightness, thus to have 25% accuracy on low brightness
72  * minimum change interval should be 100/25 * 4 * 64 = 512.
73  */
74 #define CHANGE_INTERVAL 2000 // ~ 83Hz for each tube
75 //#define CHANGE_INTERVAL 1600 // ~ 104Hz for each tube
76 //#define CHANGE_INTERVAL 800 // ~ 160Hz for each tube
77 #define DISPLAY_BRIGHTNESS_RANGE (uint32_t)( NIXIE_MAX_BRIGHTNESS - 2 )
78 
79 #define MIN_DESIRED_BRIGHTNESS ( 3 * NIXIE_MAX_BRIGHTNESS / 32 )
80 #define MIN_NIXIE_IMPULSE_SPEC ( 70 ) // 70us per spec
81 #if CHANGE_INTERVAL * MIN_DESIRED_BRIGHTNESS / NIXIE_MAX_BRIGHTNESS < MIN_NIXIE_IMPULSE_SPEC
82  #error "Minimum nixie impulse should be at least 70us. Please increase change interval"
83 #endif

88 // Do not use lower bit for invisible states
89 // Lower bit=1 means that tube is burning
90 typedef enum
91 {
92  TUBE_OFF = 0x00, // invisible
93  TUBE_NORMAL = 0x01, // visible
94  TUBE_SCROLL_WRAP = 0x03, // visible
95  TUBE_SCROLL_FORWARD = 0x05, // visible
96  TUBE_SCROLL_BACK = 0x07, // visible
97  TUBE_SMOOTH_OFF = 0x09, // visible
98  TUBE_BLINK = 0x0B, // visible
99  TUBE_BLINK_ON = TUBE_BLINK, // 0x0B visible
100  TUBE_BLINK_OFF = 0x0C, // invisible
101  TUBE_SCROLL_OFF = 0x0D, // visible
102  TUBE_SCROLL_ON = 0x0E, // invisible
103 } ENixieTubeState;
