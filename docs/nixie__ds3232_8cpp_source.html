29 bool Ds3231::init()
30 {
31  /* Init DS3232/DS3231 */
32  Wire.beginTransmission(I2C_ADDR_DS3231);
33  Wire.write(0x0E); // Write to control register
34  Wire.write(B00011100);
35  Wire.write(B00110000);
36  if ( Wire.endTransmission() != 0)
37  {
38  m_no_device = true;
39  m_seconds = 0;
40  m_minutes = 0;
41  m_hours = 0;
42  m_day_of_week = 1;
43  m_day = 1;
44  m_month = 1;
45  m_year = 16;
46  }
47  else
48  {
49  getDateTime();
50  }
51  return !m_no_device;
52 }

69 void Ds3231::getDateTime()
70 {
71  if (m_no_device)
72  {
73  return;
74  }
75  Wire.beginTransmission(I2C_ADDR_DS3231);
76  Wire.write(0x00);
77  Wire.endTransmission();
78  Wire.requestFrom(I2C_ADDR_DS3231, 7);
79 
80  if (Wire.available())

82  m_seconds = Wire.read();

87  m_month = Wire.read() & 0x1F;
88  m_year = Wire.read();
89  m_lastRefreshTs = g_nixieMs;

91 }

142 int16_t Ds3231::getTemp()
143 {
144  if (m_no_device)
145  {
146  return 0;
147  }
148  //temp registers (11h-12h) get updated automatically every 64s
149  Wire.beginTransmission(I2C_ADDR_DS3231);
150  Wire.write(0x11);
151  Wire.endTransmission();
152  Wire.requestFrom(I2C_ADDR_DS3231, 2);
153  if(Wire.available())
154  {
155  uint8_t temp = Wire.read();
156  return (((temp & 0x7F) << 2) | (Wire.read() >> 6)) * ((temp & 0x80) ? -1: 1);
157  }
158  else
159  return 0;
160 }

186 int16_t Ds3231::timeDelta(int16_t min1, int16_t min2)
187 {
188  int16_t delta1 = min2 - min1;
189  int16_t delta2;
190  if (delta1 < 0)
191  {
192  delta1 = - delta1;
193  delta2 = 24*60 + min2 - min1;
194  }
195  else
196  {
197  delta2 = 24*60 + min1 - min2;
198  }
199  return delta1 < delta2 ? delta1: delta2;
200 }

203 void Ds3231::refreshTime(ERefreshType type)
204 {
205  if ( DS3231_REFRESH_FORCED == type )
206  {
207  getDateTime();
208  return;
209  }
210  /* Updating RTC to reduce requests to RTC chip */
211  if ((uint16_t)(g_nixieMs - m_lastRefreshTs) > 1000)
212  {
213  if ((uint16_t)(g_nixieMs - m_lastRefreshTs) > 3000)
214  {
215  getDateTime();
216  }
217  else
218  {
219  uint8_t seconds = Ds3231::toDecimal(m_seconds) + 1;
220  if (seconds > 59)
221  {
222  getDateTime();
223  }
224  else
225  m_seconds = Ds3231::toInternal(seconds);
226  m_lastRefreshTs += 1000;
227  }
228  }
229 }
