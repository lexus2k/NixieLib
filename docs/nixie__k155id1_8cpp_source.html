25 void K155ID1::switchPin(uint8_t pin)
26 {
27  /* We should not set binary code out of bound. */
28  /* Otherwise, this can break K155ID1 driver. */
29  if (pin < 10)
30  {
31  for (uint8_t i=0; i<4; i++)
32  {
33  if ((pin >> i) & 0x01)
34  NixieOs::pinHigh(m_pins[i]);
35  else
36  NixieOs::pinLow(m_pins[i]);
37  }
38  }
39  else
40  {
41  for (uint8_t i=0; i<4; i++)
42  {
43  NixieOs::pinHigh(m_pins[i]);
44  }
45  }
46 }
